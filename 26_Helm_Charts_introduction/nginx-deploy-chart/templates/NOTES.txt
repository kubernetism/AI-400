1. Get the application status:
   kubectl get all -n {{ .Release.Namespace }} -l app.kubernetes.io/instance={{ .Release.Name }}

2. To access the nginx service:
{{- if eq .Values.service.type "ClusterIP" }}
   kubectl port-forward svc/{{ include "nginx-deploy-chart.fullname" . }} 8080:{{ .Values.service.port }} -n {{ .Release.Namespace }}
   echo "Visit http://127.0.0.1:8080"
{{- else if eq .Values.service.type "NodePort" }}
   echo "NodePort service exposed on node port $(kubectl get svc {{ include "nginx-deploy-chart.fullname" . }} -n {{ .Release.Namespace }} -o jsonpath='{.spec.ports[0].nodePort}')"
{{- else }}
   echo "Check external load balancer ingress with kubectl get svc {{ include "nginx-deploy-chart.fullname" . }} -n {{ .Release.Namespace }}"
{{- end }}
